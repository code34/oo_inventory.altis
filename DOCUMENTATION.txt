	/*
		Function: _inventory = ["new", []] call OO_INVENTORY;
		create a new OO_INVENTORY object
		
		Parameters:
			array - empty array (or old inventory if you want to restore from a media or other)
	*/


	/*
		Function: = ["load", player] call _inventory;
		Load inventory on the object
		
		Parameters:
			- object - object who will load the inventory
	*/

	/*
		Function: = "getInventory" call _inventory;
		Return an array containing the inventory
		
		Parameters:
			none
	*/

	/*
		Function: = ["setInventory", array] call _inventory;
		Set the contents of inventory
		
		Parameters:
			array - set the inventory with array
	*/	

	/*
		Function: = ["save", player] call _inventory;
		Save the inventory from the object
		
		Parameters:
			- object - objet source from we will save the inventory
	*/


	/*
		Function: = ["clear", player] call _inventory;
		clear the inventory of object
		
		Parameters:
			- object - objet who will clear the inventory
	*/	

	/*
		Function: = ["delete", _inventory] call OO_INVENTORY;
		delete the OO_INVENTORY object
		
		Parameters:
			OO_INVENTORY object to delete
	*/
